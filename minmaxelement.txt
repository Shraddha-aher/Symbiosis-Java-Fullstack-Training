package demo;

public class MinMaxElement {

				public static void main(String[] args) {
		        int[] array = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3}; // Example array
		        Result result = findMinMax(array);
		        
		        System.out.println("Minimum Element: " + result.minValue + " at index " + result.minIndex);
		        System.out.println("Maximum Element: " + result.maxValue + " at index " + result.maxIndex);
		    }

		    public static Result findMinMax(int[] array) {
		        if (array.length == 0) {
		            throw new IllegalArgumentException("Array cannot be empty.");
		        }
		        
		        int minValue = array[0];
		        int maxValue = array[0];
		        int minIndex = 0;
		        int maxIndex = 0;
		        
		        for (int i = 1; i < array.length; i++) {
		            if (array[i] < minValue) {
		                minValue = array[i];
		                minIndex = i;
		            }
		            if (array[i] > maxValue) {
		                maxValue = array[i];
		                maxIndex = i;
		            }
		        }
		        
		        return new Result(minValue, minIndex, maxValue, maxIndex);
		    }

		    static class Result {
		        int minValue;
		        int minIndex;
		        int maxValue;
		        int maxIndex;

		        Result(int minValue, int minIndex, int maxValue, int maxIndex) {
		            this.minValue = minValue;
		            this.minIndex = minIndex;
		            this.maxValue = maxValue;
		            this.maxIndex = maxIndex;
		        }
		    }
		

		// TODO Auto-generated method stub

	}
